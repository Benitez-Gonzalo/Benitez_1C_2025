106static void measuringHardnessFunction(void *pvParameter){
107 while (true) {
108 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
109 // Leer valor del ADC
110 AnalogInputReadSingle(CH0, &adc_raw);
111
112 // Convertir a voltaje
113 v_adc = ((float)adc_raw/1000);
114
115 // Calcular si se está en un voltaje válido
116 if (v_adc < 3.3) {
117 r_water = ((v_adc * R1) / (3.3 - v_adc))*2.83; //2,83 es el valor de calibración. Se obtuvo experimentalmente.
118 conductivity = (2.27 / r_water) * 1000000; // Convertir a µS/cm. Constante de celda: 2,27 [1/cm]
119 water_hardness = CONVERSION_FACTOR*conductivity;
120
121 uint8_t new_led_state = (water_hardness > UMBRAL) ? 1 : 0;
122
123 if (new_led_state != led_state) {
124 if (new_led_state) {
125 LedOn(LED_1);
126 } else {
127 LedOff(LED_1);
128 }
129 led_state = new_led_state;
130 }
131 }
132 UartSendString(UART_PC, buffer);
133 }
134}
